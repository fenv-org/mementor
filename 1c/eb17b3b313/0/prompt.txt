Event loop pattern:
```rust
loop {
    terminal.draw(|frame| self.render(frame))?;
    
    if event::poll(std::time::Duration::from_millis(100))? {
        if let Event::Key(key) = event::read()? {
            self.handle_key(key).await;
        }
    }
    
    if !self.running {
        break;
    }
}
```
